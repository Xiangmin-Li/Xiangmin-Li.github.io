```csharp
class LogManager
{
    GameEventManager GameEventManager;
    int EnemyDie = 0;
    public LogManager(GameEventManager _GameEventManager)
    {
        GameEventManager = _GameEventManager;
        GameEventManager.OnEnemySpawned += Logger;
        GameEventManager.OnEnemyDefeated += LoggerDie;
        GameEventManager.OnEnemyDefeated += AchievementSystem;
    }

    void Logger(object? sender, EnemyInfo enemyInfo)//输出日志
    {
        Console.WriteLine($"生成怪物:{enemyInfo.enemyName}");
    }
    void LoggerDie(object? sender, EnemyInfo enemyInfo)//输出日志
    {
        Console.WriteLine($"{enemyInfo.enemyName}已被击杀");
    }
    void AchievementSystem(object? sender, EnemyInfo enemyInfo)//记录击杀数
    {
        EnemyDie++;
        Console.WriteLine($"当前击杀数:{EnemyDie}");
    }
}
```
